270 static void adapter_state_changed_cbx (int result, bt_adapter_state_e adapter_state, void * user_data)
271 {
272  BluetoothExtends * this = (BluetoothExtends *)user_data;
273 
274  if ( result != BT_ERROR_NONE )
275  {
276  this->accessible = false;
277  return;
278  }
279  if ( adapter_state == BT_ADAPTER_ENABLED )
280  {
281  bt_adapter_visibility_mode_e mode;
282 
283  int duration = 1;
284 
285  bt_adapter_get_visibility (&mode, &duration);
286 
287  switch (mode)
288  {
289  case BT_ADAPTER_VISIBILITY_MODE_NON_DISCOVERABLE:
290  this->accessible = false;
291  break;
292  case BT_ADAPTER_VISIBILITY_MODE_GENERAL_DISCOVERABLE:
293  this->accessible = true;
294  break;
295  case BT_ADAPTER_VISIBILITY_MODE_LIMITED_DISCOVERABLE:
296  this->accessible = true;
297  break;
298  }
299  }
300  else
301  {
302  this->accessible = false;
303  }
304 }
