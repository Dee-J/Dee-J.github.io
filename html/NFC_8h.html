NFC NewNFC (void)
{
NFCExtends * this = (NFCExtends *)malloc (sizeof (NFCExtends));
this->nfc.isAccessible = isNFCAccessible;
this->nfc.onConnect = onNFCConnect;
this->nfc.onDisconnect = onNFCDisconnect;
this->nfc.Send = NFCSend;
this->nfc.Recv = NFCRecv;
return &this->nfc;
}
